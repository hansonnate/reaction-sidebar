Organizations
    uri: /organizations
    methods:
        GET - gets a list of all organizations
        POST - posts an organization
    
    uri: /organizations/{id}
    methods:
        GET - get an organization by id
        PATCH - update an organization by id
        DELETE - delete an organization by id
